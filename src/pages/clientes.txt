import { useState, useEffect } from 'react'
import { 
  Users, 
  Plus, 
  Search, 
  Phone, 
  Mail, 
  MapPin, 
  Edit2, 
  Trash2, 
  X,
  CheckCircle,
  AlertTriangle,
  ShoppingBag,
  History,
  Filter,
  Download,
  UserPlus
} from 'lucide-react'

const API_URL = 'http://localhost:3000/api'

export default function Clientes() {
  const [clientes, setClientes] = useState([])
  const [clientesFiltrados, setClientesFiltrados] = useState([])
  const [cargando, setCargando] = useState(false)
  const [mensaje, setMensaje] = useState('')
  const [busqueda, setBusqueda] = useState('')
  const [filtroActivo, setFiltroActivo] = useState('todos') // todos, conCompras, sinCompras
  
  // Estados para formulario
  const [mostrarFormulario, setMostrarFormulario] = useState(false)
  const [clienteEditando, setClienteEditando] = useState(null)
  const [nuevoCliente, setNuevoCliente] = useState({
    nombre: '',
    telefono: '',
    email: '',
    direccion: '',
    rfc: '',
    tipo: 'regular', // regular, mayorista, vip
    notas: ''
  })

  // Estado para historial de compras
  const [clienteSeleccionado, setClienteSeleccionado] = useState(null)
  const [historialCompras, setHistorialCompras] = useState([])
  const [mostrarHistorial, setMostrarHistorial] = useState(false)

  useEffect(() => {
    cargarClientes()
  }, [])

  useEffect(() => {
    filtrarClientes()
  }, [busqueda, filtroActivo, clientes])

  const cargarClientes = async () => {
    setCargando(true)
    try {
      const response = await fetch(`${API_URL}/clientes`)
      const data = await response.json()
      setClientes(data)
      setMensaje('')
    } catch (error) {
      setMensaje('❌ Error al cargar clientes')
      console.error('Error:', error)
    } finally {
      setCargando(false)
    }
  }

  const filtrarClientes = () => {
    let filtrados = [...clientes]

    // Filtro por búsqueda
    if (busqueda.trim()) {
      const termino = busqueda.toLowerCase()
      filtrados = filtrados.filter(c => 
        c.nombre?.toLowerCase().includes(termino) ||
        c.telefono?.includes(termino) ||
        c.email?.toLowerCase().includes(termino) ||
        c.rfc?.toLowerCase().includes(termino)
      )
    }

    // Filtro por tipo de cliente
    if (filtroActivo === 'conCompras') {
      filtrados = filtrados.filter(c => c.totalCompras > 0)
    } else if (filtroActivo === 'sinCompras') {
      filtrados = filtrados.filter(c => !c.totalCompras || c.totalCompras === 0)
    } else if (filtroActivo !== 'todos') {
      // Filtrar por tipo: regular, mayorista, vip
      filtrados = filtrados.filter(c => c.tipo === filtroActivo)
    }

    setClientesFiltrados(filtrados)
  }

  const verHistorialCompras = async (cliente) => {
    setCargando(true)
    try {
      const response = await fetch(`${API_URL}/ventas/cliente/${cliente.id}`)
      const data = await response.json()
      setHistorialCompras(data)
      setClienteSeleccionado(cliente)
      setMostrarHistorial(true)
    } catch (error) {
      setMensaje('❌ Error al cargar historial')
    } finally {
      setCargando(false)
    }
  }

  const guardarCliente = async (e) => {
    e.preventDefault()
    setCargando(true)

    try {
      const url = clienteEditando 
        ? `${API_URL}/clientes/${clienteEditando.id}`
        : `${API_URL}/clientes`
      
      const method = clienteEditando ? 'PUT' : 'POST'
      
      const response = await fetch(url, {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(clienteEditando || nuevoCliente)
      })

      if (response.ok) {
        setMensaje(clienteEditando ? '✅ Cliente actualizado' : '✅ Cliente creado')
        cerrarFormulario()
        cargarClientes()
        setTimeout(() => setMensaje(''), 3000)
      } else {
        const error = await response.json()
        setMensaje(`❌ ${error.error || 'Error al guardar'}`)
      }
    } catch (error) {
      setMensaje('❌ Error de conexión')
    } finally {
      setCargando(false)
    }
  }

  const eliminarCliente = async (id) => {
    const cliente = clientes.find(c => c.id === id)
    if (cliente?.totalCompras > 0) {
      if (!confirm('⚠️ Este cliente tiene compras registradas. ¿Eliminar de todas formas? Los registros de ventas se mantendrán pero quedarán sin cliente asociado.')) {
        return
      }
    } else {
      if (!confirm('¿Estás seguro de eliminar este cliente?')) return
    }

    setCargando(true)
    try {
      const response = await fetch(`${API_URL}/clientes/${id}`, {
        method: 'DELETE'
      })

      if (response.ok) {
        setMensaje('✅ Cliente eliminado')
        cargarClientes()
        if (clienteSeleccionado?.id === id) {
          setMostrarHistorial(false)
          setClienteSeleccionado(null)
        }
      } else {
        setMensaje('❌ Error al eliminar')
      }
    } catch (error) {
      setMensaje('❌ Error de conexión')
    } finally {
      setCargando(false)
    }
  }

  const iniciarEdicion = (cliente) => {
    setClienteEditando(cliente)
    setMostrarFormulario(true)
    setMostrarHistorial(false)
  }

  const iniciarNuevo = () => {
    setClienteEditando(null)
    setNuevoCliente({
      nombre: '',
      telefono: '',
      email: '',
      direccion: '',
      rfc: '',
      tipo: 'regular',
      notas: ''
    })
    setMostrarFormulario(true)
    setMostrarHistorial(false)
  }

  const cerrarFormulario = () => {
    setMostrarFormulario(false)
    setClienteEditando(null)
  }

  const cerrarHistorial = () => {
    setMostrarHistorial(false)
    setClienteSeleccionado(null)
    setHistorialCompras([])
  }

  const getTipoColor = (tipo) => {
    switch (tipo) {
      case 'vip': return 'bg-purple-100 text-purple-800'
      case 'mayorista': return 'bg-blue-100 text-blue-800'
      case 'regular': return 'bg-gray-100 text-gray-800'
      default: return 'bg-gray-100 text-gray-800'
    }
  }

  const getTipoLabel = (tipo) => {
    switch (tipo) {
      case 'vip': return 'VIP'
      case 'mayorista': return 'Mayorista'
      case 'regular': return 'Regular'
      default: return 'Regular'
    }
  }

  const exportarClientes = () => {
    const dataStr = JSON.stringify(clientes, null, 2)
    const dataUri = 'data:application/json;charset=utf-8,'+ encodeURIComponent(dataStr)
    const exportFileDefaultName = `clientes_${new Date().toISOString().split('T')[0]}.json`
    const linkElement = document.createElement('a')
    linkElement.setAttribute('href', dataUri)
    linkElement.setAttribute('download', exportFileDefaultName)
    linkElement.click()
  }

  // Estadísticas
  const stats = {
    total: clientes.length,
    conCompras: clientes.filter(c => c.totalCompras > 0).length,
    vip: clientes.filter(c => c.tipo === 'vip').length,
    mayoristas: clientes.filter(c => c.tipo === 'mayorista').length
  }

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex flex-col md:flex-row justify-between items-start md:items-center gap-4">
        <div>
          <h1 className="text-2xl font-bold text-gray-900 flex items-center">
            <Users className="w-8 h-8 mr-3 text-orange-600" />
            Gestión de Clientes
          </h1>
          <p className="text-gray-500 mt-1">Administra tu base de clientes y su historial</p>
        </div>
        <div className="flex gap-2">
          <button
            onClick={exportarClientes}
            className="bg-gray-100 text-gray-700 px-4 py-2 rounded-lg hover:bg-gray-200 flex items-center"
            title="Exportar JSON"
          >
            <Download className="w-4 h-4 mr-2" />
            Exportar
          </button>
          <button
            onClick={iniciarNuevo}
            className="bg-orange-600 text-white px-4 py-2 rounded-lg hover:bg-orange-700 flex items-center"
          >
            <UserPlus className="w-5 h-5 mr-2" />
            Nuevo Cliente
          </button>
        </div>
      </div>

      {/* Mensaje */}
      {mensaje && (
        <div className={`p-4 rounded-lg text-center font-medium ${
          mensaje.includes('✅') ? 'bg-green-100 text-green-800 border border-green-300' :
          'bg-red-100 text-red-800 border border-red-300'
        }`}>
          {mensaje}
        </div>
      )}

      {/* Estadísticas */}
      <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
        <div className="bg-blue-50 rounded-lg p-4">
          <p className="text-sm text-blue-600 font-medium">Total Clientes</p>
          <p className="text-2xl font-bold text-blue-900">{stats.total}</p>
        </div>
        <div className="bg-green-50 rounded-lg p-4">
          <p className="text-sm text-green-600 font-medium">Con Compras</p>
          <p className="text-2xl font-bold text-green-900">{stats.conCompras}</p>
        </div>
        <div className="bg-purple-50 rounded-lg p-4">
          <p className="text-sm text-purple-600 font-medium">Clientes VIP</p>
          <p className="text-2xl font-bold text-purple-900">{stats.vip}</p>
        </div>
        <div className="bg-orange-50 rounded-lg p-4">
          <p className="text-sm text-orange-600 font-medium">Mayoristas</p>
          <p className="text-2xl font-bold text-orange-900">{stats.mayoristas}</p>
        </div>
      </div>

      {/* Filtros y Búsqueda */}
      <div className="bg-white rounded-lg shadow-md p-4 space-y-4">
        <div className="flex flex-col md:flex-row gap-4">
          <div className="flex-1 relative">
            <Search className="w-5 h-5 absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400" />
            <input
              type="text"
              placeholder="Buscar por nombre, teléfono, email o RFC..."
              value={busqueda}
              onChange={(e) => setBusqueda(e.target.value)}
              className="w-full pl-10 pr-4 py-2 border rounded-lg focus:ring-2 focus:ring-orange-500 focus:border-orange-500"
            />
          </div>
          <div className="flex items-center gap-2">
            <Filter className="w-5 h-5 text-gray-500" />
            <select
              value={filtroActivo}
              onChange={(e) => setFiltroActivo(e.target.value)}
              className="border rounded-lg px-3 py-2 focus:ring-2 focus:ring-orange-500"
            >
              <option value="todos">Todos los clientes</option>
              <option value="conCompras">Con compras</option>
              <option value="sinCompras">Sin compras</option>
              <option value="vip">VIP</option>
              <option value="mayorista">Mayoristas</option>
              <option value="regular">Regulares</option>
            </select>
          </div>
        </div>
      </div>

      {/* Formulario Crear/Editar */}
      {mostrarFormulario && (
        <div className="bg-white rounded-lg shadow-md p-6 border-2 border-orange-200">
          <div className="flex justify-between items-center mb-4">
            <h2 className="text-lg font-bold">
              {clienteEditando ? 'Editar Cliente' : 'Nuevo Cliente'}
            </h2>
            <button onClick={cerrarFormulario} className="text-gray-500 hover:text-gray-700">
              <X className="w-5 h-5" />
            </button>
          </div>
          
          <form onSubmit={guardarCliente} className="space-y-4">
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">Nombre completo *</label>
                <input
                  type="text"
                  required
                  value={clienteEditando ? clienteEditando.nombre : nuevoCliente.nombre}
                  onChange={(e) => {
                    const val = e.target.value
                    clienteEditando 
                      ? setClienteEditando({...clienteEditando, nombre: val})
                      : setNuevoCliente({...nuevoCliente, nombre: val})
                  }}
                  className="w-full border rounded-lg px-3 py-2"
                  placeholder="Nombre del cliente"
                />
              </div>
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">Tipo de cliente</label>
                <select
                  value={clienteEditando ? clienteEditando.tipo : nuevoCliente.tipo}
                  onChange={(e) => {
                    const val = e.target.value
                    clienteEditando 
                      ? setClienteEditando({...clienteEditando, tipo: val})
                      : setNuevoCliente({...nuevoCliente, tipo: val})
                  }}
                  className="w-full border rounded-lg px-3 py-2"
                >
                  <option value="regular">Regular</option>
                  <option value="mayorista">Mayorista</option>
                  <option value="vip">VIP</option>
                </select>
              </div>
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">Teléfono</label>
                <div className="relative">
                  <Phone className="w-4 h-4 absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400" />
                  <input
                    type="tel"
                    value={clienteEditando ? clienteEditando.telefono : nuevoCliente.telefono}
                    onChange={(e) => {
                      const val = e.target.value
                      clienteEditando 
                        ? setClienteEditando({...clienteEditando, telefono: val})
                        : setNuevoCliente({...nuevoCliente, telefono: val})
                    }}
                    className="w-full pl-10 border rounded-lg px-3 py-2"
                    placeholder="(123) 456-7890"
                  />
                </div>
              </div>
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">Email</label>
                <div className="relative">
                  <Mail className="w-4 h-4 absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400" />
                  <input
                    type="email"
                    value={clienteEditando ? clienteEditando.email : nuevoCliente.email}
                    onChange={(e) => {
                      const val = e.target.value
                      clienteEditando 
                        ? setClienteEditando({...clienteEditando, email: val})
                        : setNuevoCliente({...nuevoCliente, email: val})
                    }}
                    className="w-full pl-10 border rounded-lg px-3 py-2"
                    placeholder="cliente@email.com"
                  />
                </div>
              </div>
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">RFC</label>
                <input
                  type="text"
                  value={clienteEditando ? clienteEditando.rfc : nuevoCliente.rfc}
                  onChange={(e) => {
                    const val = e.target.value.toUpperCase()
                    clienteEditando 
                      ? setClienteEditando({...clienteEditando, rfc: val})
                      : setNuevoCliente({...nuevoCliente, rfc: val})
                  }}
                  className="w-full border rounded-lg px-3 py-2 uppercase"
                  placeholder="XAXX010101000"
                />
              </div>
              <div className="md:col-span-2">
                <label className="block text-sm font-medium text-gray-700 mb-1">Dirección</label>
                <div className="relative">
                  <MapPin className="w-4 h-4 absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400" />
                  <input
                    type="text"
                    value={clienteEditando ? clienteEditando.direccion : nuevoCliente.direccion}
                    onChange={(e) => {
                      const val = e.target.value
                      clienteEditando 
                        ? setClienteEditando({...clienteEditando, direccion: val})
                        : setNuevoCliente({...nuevoCliente, direccion: val})
                    }}
                    className="w-full pl-10 border rounded-lg px-3 py-2"
                    placeholder="Dirección completa"
                  />
                </div>
              </div>
              <div className="md:col-span-2">
                <label className="block text-sm font-medium text-gray-700 mb-1">Notas</label>
                <textarea
                  rows="3"
                  value={clienteEditando ? clienteEditando.notas : nuevoCliente.notas}
                  onChange={(e) => {
                    const val = e.target.value
                    clienteEditando 
                      ? setClienteEditando({...clienteEditando, notas: val})
                      : setNuevoCliente({...nuevoCliente, notas: val})
                  }}
                  className="w-full border rounded-lg px-3 py-2"
                  placeholder="Información adicional sobre el cliente..."
                />
              </div>
            </div>

            <div className="flex justify-end gap-2 pt-4">
              <button
                type="button"
                onClick={cerrarFormulario}
                className="px-6 py-2 border border-gray-300 rounded-lg hover:bg-gray-50"
              >
                Cancelar
              </button>
              <button
                type="submit"
                disabled={cargando}
                className="bg-orange-600 text-white px-6 py-2 rounded-lg hover:bg-orange-700 disabled:opacity-50"
              >
                {cargando ? 'Guardando...' : (clienteEditando ? 'Actualizar' : 'Guardar')}
              </button>
            </div>
          </form>
        </div>
      )}

      {/* Historial de Compras Modal */}
      {mostrarHistorial && clienteSeleccionado && (
        <div className="bg-white rounded-lg shadow-md p-6 border-2 border-blue-200">
          <div className="flex justify-between items-center mb-4">
            <div>
              <h2 className="text-lg font-bold flex items-center">
                <History className="w-5 h-5 mr-2 text-blue-600" />
                Historial de Compras
              </h2>
              <p className="text-sm text-gray-500">{clienteSeleccionado.nombre}</p>
            </div>
            <button onClick={cerrarHistorial} className="text-gray-500 hover:text-gray-700">
              <X className="w-5 h-5" />
            </button>
          </div>

          {historialCompras.length === 0 ? (
            <div className="text-center py-8 text-gray-500">
              <ShoppingBag className="w-12 h-12 mx-auto mb-3 text-gray-300" />
              <p>Este cliente aún no tiene compras registradas</p>
            </div>
          ) : (
            <div className="overflow-x-auto">
              <table className="w-full">
                <thead className="bg-gray-50">
                  <tr>
                    <th className="px-4 py-3 text-left text-sm font-medium text-gray-700">Fecha</th>
                    <th className="px-4 py-3 text-left text-sm font-medium text-gray-700">Productos</th>
                    <th className="px-4 py-3 text-right text-sm font-medium text-gray-700">Total</th>
                    <th className="px-4 py-3 text-center text-sm font-medium text-gray-700">Estado</th>
                  </tr>
                </thead>
                <tbody className="divide-y divide-gray-200">
                  {historialCompras.map((venta) => (
                    <tr key={venta.id} className="hover:bg-gray-50">
                      <td className="px-4 py-3 text-sm text-gray-900">
                        {new Date(venta.fecha).toLocaleDateString()}
                      </td>
                      <td className="px-4 py-3 text-sm text-gray-600">
                        {venta.items?.length || 0} productos
                      </td>
                      <td className="px-4 py-3 text-sm text-gray-900 text-right font-medium">
                        ${venta.total?.toFixed(2)}
                      </td>
                      <td className="px-4 py-3 text-center">
                        <span className={`px-2 py-1 rounded-full text-xs ${
                          venta.estado === 'completada' ? 'bg-green-100 text-green-800' :
                          venta.estado === 'pendiente' ? 'bg-yellow-100 text-yellow-800' :
                          'bg-gray-100 text-gray-800'
                        }`}>
                          {venta.estado || 'completada'}
                        </span>
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
              <div className="mt-4 p-4 bg-gray-50 rounded-lg flex justify-between items-center">
                <span className="font-medium text-gray-700">Total comprado:</span>
                <span className="text-xl font-bold text-green-600">
                  ${historialCompras.reduce((sum, v) => sum + (v.total || 0), 0).toFixed(2)}
                </span>
              </div>
            </div>
          )}
        </div>
      )}

      {/* Lista de Clientes */}
      <div className="bg-white rounded-lg shadow-md overflow-hidden">
        <div className="p-4 bg-gray-50 border-b flex justify-between items-center">
          <h2 className="text-lg font-bold text-gray-800 flex items-center">
            <Users className="w-5 h-5 mr-2" />
            Lista de Clientes ({clientesFiltrados.length})
          </h2>
        </div>

        {clientesFiltrados.length === 0 ? (
          <div className="p-8 text-center text-gray-500">
            <Users className="w-16 h-16 mx-auto mb-4 text-gray-300" />
            <p className="text-lg">No se encontraron clientes</p>
            <p className="text-sm mt-1">Intenta con otra búsqueda o crea uno nuevo</p>
          </div>
        ) : (
          <div className="overflow-x-auto">
            <table className="w-full">
              <thead className="bg-gray-100">
                <tr>
                  <th className="px-4 py-3 text-left text-sm font-medium text-gray-700">Cliente</th>
                  <th className="px-4 py-3 text-left text-sm font-medium text-gray-700">Contacto</th>
                  <th className="px-4 py-3 text-center text-sm font-medium text-gray-700">Tipo</th>
                  <th className="px-4 py-3 text-center text-sm font-medium text-gray-700">Compras</th>
                  <th className="px-4 py-3 text-right text-sm font-medium text-gray-700">Total Gastado</th>
                  <th className="px-4 py-3 text-center text-sm font-medium text-gray-700">Acciones</th>
                </tr>
              </thead>
              <tbody className="divide-y divide-gray-200">
                {clientesFiltrados.map((cliente) => (
                  <tr key={cliente.id} className="hover:bg-gray-50">
                    <td className="px-4 py-3">
                      <div className="flex items-center">
                        <div className="w-10 h-10 rounded-full bg-orange-100 flex items-center justify-center mr-3">
                          <span className="text-orange-600 font-bold">
                            {cliente.nombre?.charAt(0).toUpperCase()}
                          </span>
                        </div>
                        <div>
                          <p className="font-medium text-gray-900">{cliente.nombre}</p>
                          {cliente.rfc && (
                            <p className="text-xs text-gray-500">RFC: {cliente.rfc}</p>
                          )}
                        </div>
                      </div>
                    </td>
                    <td className="px-4 py-3">
                      <div className="space-y-1">
                        {cliente.telefono && (
                          <p className="text-sm text-gray-600 flex items-center">
                            <Phone className="w-3 h-3 mr-1" /> {cliente.telefono}
                          </p>
                        )}
                        {cliente.email && (
                          <p className="text-sm text-gray-600 flex items-center">
                            <Mail className="w-3 h-3 mr-1" /> {cliente.email}
                          </p>
                        )}
                      </div>
                    </td>
                    <td className="px-4 py-3 text-center">
                      <span className={`px-2 py-1 rounded-full text-xs font-medium ${getTipoColor(cliente.tipo)}`}>
                        {getTipoLabel(cliente.tipo)}
                      </span>
                    </td>
                    <td className="px-4 py-3 text-center">
                      <span className={`text-sm font-medium ${cliente.totalCompras > 0 ? 'text-green-600' : 'text-gray-400'}`}>
                        {cliente.totalCompras || 0}
                      </span>
                    </td>
                    <td className="px-4 py-3 text-right">
                      <span className="font-medium text-gray-900">
                        ${(cliente.totalGastado || 0).toFixed(2)}
                      </span>
                    </td>
                    <td className="px-4 py-3">
                      <div className="flex justify-center gap-2">
                        <button
